#include <Servo.h>

Servo servo1;
Servo servo2;

void setup() {
  servo1.attach(5);   // Servo 1 pin
  servo2.attach(6);   // Servo 2 pin
}

void loop() {
  // Move in opposite directions
  for (int pos = 0; pos <= 90; pos++) {
    servo1.write(pos);         // anticlockwise
    servo2.write(180 - pos);   // clockwise
    delay(15);
  }

  delay(500); // pause

  // Move back to start
  for (int pos = 90; pos >= 0; pos--) {
    servo1.write(pos);
    servo2.write(180 - pos);
    delay(15);
  }

  delay(500); // pause
}
